# Civil War
country_event = {
	id = france.12
	title = france.12.t
	desc = france.12.d
	picture = GFX_report_event_fra_first_coalition

	fire_only_once = yes

	trigger = {
		NOT = { has_global_flag = scw_over }
		tag = FRA
		NOT = { has_government = democratic }
		has_completed_focus = fra_flee
	}

	mean_time_to_happen = {
		days = 1
	}

	option = { # Uh Oh
		name = france.12.a
		trigger = { is_ai = no }
		hidden_effect = {
			set_capital = 735
			start_civil_war = {
				ruling_party = fascism
				ideology = democratic
				size = 0.5
				capital = 16
				states = { 16 29 18 28 17 27 20 21 32 22 31 26 24 15 14 30 33 19 25 23 }
			}
			random_country = {
				limit = {
					original_tag = FRA
					has_government = democratic
				}
				load_oob = "fra_civil_war"
				set_equipment_fraction = 0.7
				set_stability = 0.9
				add_popularity = {
					ideology = democratic
					popularity = 1
				}
			}
			news_event = { hours = 6 id = news.99 }

			load_oob = "frr_civil_war"
			set_equipment_fraction = 1
			set_stability = 0.9

			add_popularity = {
				ideology = fascism
				popularity = 1
			}
		}
	}
}

# First Coalition War - Austria wants peace
country_event = {
	id = france.26
	title = france.26.t
	desc = france.26.d
	picture = GFX_report_event_france_first_coalition_treaty

	is_triggered_only = yes

	option = {
		name = france.26.a
		if = {
			limit = { has_war_with = PRE }
			white_peace = PRE
		}
		if = {
			limit = { has_war_with = AUS }
			white_peace = AUS
		}
		FRA = {
			transfer_state = 6
			transfer_state = 34
		}
		AUS = {
			remove_from_faction = PRE
		}
		PRE = {
			white_peace = FRA
		}
		white_peace = FRA
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = france.26.b
		ai_chance = {
			factor = 0
		}
		AUS = { country_event = austria.4 }
	}
}
