add_namespace = austria

# Ally France
country_event = {
	id = austria.1
	title = austria.1.t
	desc = austria.1.d
	picture = GFX_report_event_hitler_croatia_handshake

	is_triggered_only	= yes

	option = { # Support the Yugoslavs alone
		name = austria.1.a
		ai_chance = {
			factor = 99
		}
		if = {
			limit = {
				is_in_faction = yes
				is_faction_leader = yes
			}
			create_faction = "french_alliance"
			add_to_faction = FRA
			add_to_faction = PRE
			FRA = {
				add_ai_strategy = {
					type = alliance
					id = "AUS"
					value = 200
				}
			}
			PRE = {
				add_ai_strategy = {
					type = alliance
					id = "AUS"
					value = 200
				}
			}
		}
		if = {
			limit = {
				OR = {
					is_in_faction = no
					is_faction_leader = no
				}
			}
			create_faction = "french_alliance"
			add_to_faction = FRA
			add_to_faction = PRE
			FRA = {
				add_ai_strategy = {
					type = alliance
					id = "AUS"
					value = 200
				}
			}
			PRE = {
				add_ai_strategy = {
					type = alliance
					id = "AUS"
					value = 200
				}
			}
		}
	}
}

# Put king back on throne

	id = austria.5

	picture = GFX_report_event_france_first_coalition_treaty

	option = {
		name = austria.5.a
		FRA = {
			set_politics = {
				ruling_party = fascism
				elections_allowed = no
			}
			add_popularity = {
				ideology = fascism
				popularity = 1.0
			}
		}
		PRE = {
			white_peace = FRA
			remove_from_faction = PRE
		}
		white_peace = FRA
		ai_chance = {
			factor = 100
		}
	}
	option = {
		name = austria.5.b
	}
}
